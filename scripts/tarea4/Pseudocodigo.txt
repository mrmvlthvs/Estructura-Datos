3. Proceso:
    Escribir("\n--- Procesando Gramática L-System ---");
    c_acumulada = "FX";
    
    i = 0;
    terminar mientras (i < orden) empezar
        c_acumulada = llamar a dragonEvolution(c_acumulada);
        i = i + 1;
    terminar mientras
    
    Escribir("Evolución completada. Tamaño: " + llamar a c_acumulada.length() + " caracteres.");
    Escribir("Trazando en ventana gráfica...");
    
    fr_d = longitud * llamar a Math.pow(1.4142, orden);
    origen_x = - (fr_d / 3.5);
    
    si ((orden % 2) == 0) entonces
        origen_y = (fr_d / 4.0);
    sino entonces
        origen_y = -(fr_d / 12.0);
    terminar si
    
    si (orden < 6) entonces
        origen_x = -50;
        origen_y = 0;
    terminar si
    
    llamar a t.penup();
    llamar a t.goto(origen_x, origen_y);
    llamar a t.pendown();
    
    llamar a interpretDragon(c_acumulada);

3.1 dragonEvolution:String :
    Recibir atributo 'cadena_actual'
    resultado = "";
    i = 0;
    terminar mientras (i < llamar a cadena_actual.length()) empezar
        caracter = llamar a cadena_actual.charAt(i);
        segun (caracter) hacer
            caso 'X':
                resultado = resultado + "X+YF+";
                romper;
            caso 'Y':
                resultado = resultado + "-FX-Y";
                romper;
            por defecto:
                resultado = resultado + caracter;
                romper;
        terminar segun
        i = i + 1;
    terminar mientras
    retornar resultado;

3.2 interpretDragon:
    Recibir atributo 'cadena_final'
    colores = ["blue", "red", "green", "black", "yellow", "magenta"];
    dr = 0;
    i = 0;
    
    terminar mientras (i < llamar a cadena_final.length()) empezar
        caracter = llamar a cadena_final.charAt(i);
        segun (caracter) hacer
            caso '+':
                llamar a t.left(90);
                romper;
            caso '-':
                llamar a t.right(90);
                romper;
            caso 'F':
                llamar a t.pencolor(colores[dr]);
                llamar a t.forward(longitud);
                dr = (dr + 1) % 6;
                romper;
        terminar segun
        i = i + 1;
    terminar mientras
