1. meta():

   Para capturar y buscar datos de personas almacenadas en una

2.1 data():
   Persona grupo[] = new Persona[25];
   String nombre, control, fNac;
   double peso, talla;
   int cuenta = 0;

   Escribir("Deme el nombre:");
   nombre = ?;
   mientras (nombre != "fin" y cuenta < 25) empezar
      Escribir("De # de control:");
      control = ?;
      Escribir("De Fecha de Nacimiento:");
      fNac = ?;
      Escribir("Peso de la persona:");
      peso = ?;
      Escribir("Estatura de la persona:");
      talla = ?;
      grupo[cuenta] =
         nueva Persona(nombre, control, fNac, peso, talla);
      cuenta = cuenta + 1;
      Escribir("Deme el nombre:");
      nombre = ?;
   terminar
   si (cuenta == 25) empezar
      Escribir("Limite del arreglo alcanzado.");
   terminar

2.2 lectura(archi):
   Persona grupo[] = new Persona[25];
   Persona pers;
   cuenta = 0;
   in = abrir(archi);

   mientras (!eof(in) y cuenta < 25) empezar
      pers = leerRegistro(in);
      grupo[cuenta] = nueva Persona();
      grupo[cuenta].setNombre(pers.getNombre());
      grupo[cuenta].setControl(pers.getControl());
      grupo[cuenta].setFNac(pers.getFNac());
      grupo[cuenta].setPeso(pers.getPeso());
      grupo[cuenta].setTalla(pers.getTalla());
      cuenta = cuenta + 1;
   terminar
   cerrar(in);

2.3 entry(prompt):
   Escribir(prompt);

3. buscar(name): regresa un entero
   encontrado = false;
   compara = 0;

   desde indice = 0 hasta indice < cuenta ;
   incrementar indice para
      empezar
         compara = compara + 1;
         si (name.equals(grupo[indice].getNombre()))
         empezar
            encontrado = true;
            retornar indice;
         terminar
      terminar
   retornar -1;

4.0 salida(name, pos):
   Escribir("El nombre solicitado: " + name);
   si (pos != -1) empezar
      Escribir("Esta en la posicion " + pos);
      Escribir("Su numero de control es: "
               + grupo[pos].getControl());
      Escribir("Nacio el: "
               + grupo[pos].getFNac());
      Escribir("Su peso es: "
               + grupo[pos].getPeso());
      Escribir("Su talla es: "
               + grupo[pos].getTalla());
   terminar
   de otro modo empezar
      Escribir("No fue encontrado en el grupo.");
   terminar
   Escribir("Se realizaron "
            + compara
            + " comparaciones para la busqueda.");
