    intentos = 0;

    si (cuenta == 0) empezar
        Escribir("No hay personas registradas");
        regresar;
    terminar si

    Escribir("Ingrese los apellidos de la persona a buscar");
    apellidoBuscado = ?;

    si (apellidoBuscado es nulo o vacio) empezar
        Escribir("Debe ingresar al menos un apellido");
        regresar;
    terminar si

    inferior = 0;
    superior = cuenta - 1;
    encontrado = falso;

    terminar mientras (inferior <= superior) empezar
        pivote = (inferior + superior) / 2;
        intentos = intentos + 1;

        partesActual = separarNombre(grupo[pivote]);
        apellidoActual = obtenerApellidos(partesActual);

        comparacion = comparar(apellidoBuscado, apellidoActual);

        si (comparacion == 0) empezar
            Escribir("Persona encontrada en la posicion " + (pivote + 1));
            Escribir("Numero de comparaciones: " + intentos);
            Escribir(grupo[pivote]);
            encontrado = verdadero;
            regresar;
        terminar si

        si (comparacion < 0) empezar
            superior = pivote - 1;
        de otro modo empezar
            inferior = pivote + 1;
        terminar si
    terminar mientras

    si (!encontrado) empezar
        Escribir("No se encontro la persona");
    terminar si
