3. capReal(prompt)

4. capOpcion(prompt)
    mientras (verdadero) empezar
        Escribir(prompt)
        entrada = leer de la terminal sin espacios y en mayúsculas
        si (entrada es igual a "S" o entrada es igual a "N") entonces
            retornar entrada
        de otro modo
            Escribir("Error: Opcion no valida. Ingrese estrictamente 'S' para si o 'N' para no.")
    terminar mientras

5. entradas(matrizOriginal)

        fil = llamar capInt("¿Cuántos renglones tiene su matriz?:")

    hacer
        col = llamar capInt("¿Cuántas columnas hay en su matriz?:")
    mientras (col < 0)
    
    Inicializar matrizOriginal con dimensiones [fil][col]
    
    desde i = 0; mientras i < fil; incrementar i para
        desde j = 0; mientras j < col; incrementar j para empezar
            matrizOriginal[i][j] = llamar capReal("Matriz[" + i + ", " + j + "]:")
        terminar
        
    retornar matrizOriginal
