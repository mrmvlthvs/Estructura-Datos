2.a. isNum(titulo, mensaje)
    valido = falso;
    numero = nulo;
    
    mientras (no valido) empezar
        aux = leer cadena del usuario(titulo, mensaje);
        
        si (aux es nulo) empezar
            aux = "";
        terminar si
        
        si (aux.trim() es igual a "parar" sin importar mayúsculas) empezar
            retornar nulo; 
        terminar si
        
        intentar empezar
            numero = convertir aux.trim() a entero;
            valido = verdadero;
        capturar error empezar
            Escribir("No es un número entero válido ni la palabra de parada. Intente de nuevo.");
        terminar intentar
    terminar mientras
    retornar numero;

3.1. ordenar()
    comparaciones = 0;
    movimientos = 0;
    
    si (contador == 0) empezar

        retornar;
    terminar si
    
    toSort = nuevo arreglo copiando listaDatos desde 0 hasta contador - 1;
    toSort = llamar mezclar(toSort);
    
    para i desde 0 hasta contador - 1 empezar

    Escribir("Procesando ordenamiento por mezcla...");

3.b. mezclar(array)
    si (longitud(array) <= 1) empezar
        retornar array;
    terminar si
    
    medio = longitud(array) / 2;
    izquierda = nuevo arreglo copiando array desde 0 hasta medio - 1;
    derecha = nuevo arreglo copiando array desde medio hasta longitud(array) - 1;
    
    izquierda = llamar mezclar(izquierda);
    derecha = llamar mezclar(derecha);
    
    i = 0; j = 0; k = 0;
    
    mientras (i < longitud(izquierda) y j < longitud(derecha)) empezar
        comparaciones = comparaciones + 1;
        si (izquierda[i] <= derecha[j]) empezar
            array[k] = izquierda[i];
            i = i + 1;
        de otro modo empezar
            array[k] = derecha[j];
            j = j + 1;
        terminar si
        movimientos = movimientos + 1;
        k = k + 1;
    terminar mientras
    
    mientras (i < longitud(izquierda)) empezar
        array[k] = izquierda[i];
        i = i + 1;
        k = k + 1;
        movimientos = movimientos + 1;
    terminar mientras
    
    mientras (j < longitud(derecha)) empezar
        array[k] = derecha[j];
        j = j + 1;
        k = k + 1;
        movimientos = movimientos + 1;
    terminar mientras
    
    retornar array;
