2.a. isNum(titulo, mensaje)
    valido = falso;
    numero = nulo;
    
    mientras (no valido) empezar
        aux = leer cadena del usuario(titulo, mensaje);
        
        si (aux es nulo) empezar
            aux = "";
        terminar si
        
        si (aux.trim() es igual a "parar" sin importar mayúsculas) empezar
            retornar nulo; 
        terminar si
        
        intentar empezar
            numero = convertir aux.trim() a entero;
            valido = verdadero;
        capturar error empezar
            Escribir("No es un número entero válido ni la palabra de parada. Intente de nuevo.");
        terminar intentar
    terminar mientras
    retornar numero;

3.b. naturalMerge(array, n)
    aux = nuevo arreglo de enteros con tamaño n;
    ordenado = falso;
    comparaciones = 0;
    movimientos = 0;

    mientras (no ordenado) empezar
        i = 0;
        k = 0;
        ordenado = verdadero;

        mientras (i < n) empezar
            start1 = i;
            mientras (i + 1 < n y array[i] <= array[i + 1]) empezar
                comparaciones = comparaciones + 1;
                i = i + 1;
            terminar mientras
            si (i + 1 < n) empezar
                comparaciones = comparaciones + 1;
            terminar si
            end1 = i;

            si (i + 1 >= n) empezar
                para x desde start1 hasta end1 empezar
                    aux[k] = array[x];
                    k = k + 1;
                    movimientos = movimientos + 1;
                terminar para
                romper mientras;
            terminar si
            
            ordenado = falso;
            start2 = i + 1;
            i = i + 1;

            mientras (i + 1 < n y array[i] <= array[i + 1]) empezar
                comparaciones = comparaciones + 1;
                i = i + 1;
            terminar mientras
            si (i + 1 < n) empezar
                comparaciones = comparaciones + 1;
            terminar si
            end2 = i;
            
            p = start1;
            q = start2;

            mientras (p <= end1 y q <= end2) empezar
                comparaciones = comparaciones + 1;
                si (array[p] <= array[q]) empezar
                    aux[k] = array[p];
                    p = p + 1;
                de otro modo empezar
                    aux[k] = array[q];
                    q = q + 1;
                terminar si
                movimientos = movimientos + 1;
                k = k + 1;
            terminar mientras
            
            mientras (p <= end1) empezar
                aux[k] = array[p];
                p = p + 1;
                k = k + 1;
                movimientos = movimientos + 1;
            terminar mientras
            
            mientras (q <= end2) empezar
                aux[k] = array[q];
                q = q + 1;
                k = k + 1;
                movimientos = movimientos + 1;
            terminar mientras
            
            i = i + 1;
        terminar mientras
        
        para i desde 0 hasta n - 1 empezar
            array[i] = aux[i];
            movimientos = movimientos + 1;
        terminar para
    terminar mientras
