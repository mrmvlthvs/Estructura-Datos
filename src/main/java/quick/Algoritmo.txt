2. data().
    opcion ← menuCaptura()

    segun(opcion) hacer
        caso 0:
            llamar a teclado()
        caso 1:
            llamar a archivo()
    fin segun

    si (datos ≠ nulo Y tamaño(datos) > 0) empezar
        llamar a procesoOrdenar(datos)
    fin si

2.1 teclado().
    Definir lista como lista vacía
    dato ← 0

    mientras (dato ≠ -1) empezar
        Escribir("Ingrese un dato (-1 para terminar)")
        dato ← leer entero

        si (dato ≠ -1) entonces
            agregar dato a lista
        fin si
    fin mientras

    convertir lista a arreglo datos
    Escribir("Datos capturados correctamente")

2.2 archivo().
    Seleccionar archivo de texto

    si (archivo seleccionado) empezar
        Definir lista como lista vacía

        mientras (linea ← leer archivo) ≠ nulo empezar
            linea ← limpiar(linea)

            si (linea ≠ vacia) empezar
                intentar
                    dato ← convertir entero(linea)
                    agregar dato a lista
                capturar error
                    Escribir("Dato inválido")
                fin intentar
            fin si
        fin mientras

        convertir lista a arreglo datos
        Escribir("Datos cargados correctamente")
    si_no empezar
        Escribir("No se seleccionó archivo")
    fin si

3. procesoOrdenar(datos).
    comparaciones ← 0
    intercambios ← 0
    llamar a quickSort(datos, 0, tamaño(datos)-1)

    Escribir("Comparaciones: ", comparaciones)
    Escribir("Intercambios: ", intercambios)

3.1 quickSort(arr, low, high).
    si (low < high) empezar
        pi ← partition(arr, low, high)

        llamar a quickSort(arr, low, pi - 1)
        llamar a quickSort(arr, pi + 1, high)
    fin si

3.2 partition(arr, low, high).
    pivot ← arr[high]
    i ← low - 1
    c ← 0
    ic ← 0

    para j desde low hasta high - 1 hacer
        c ← c + 1

        si (arr[j] ≤ pivot) empezar
            i ← i + 1

            intercambiar arr[i] con arr[j]
            ic ← ic + 1
        fin si
    fin para

    intercambiar arr[i+1] con arr[high]
    ic ← ic + 1

    comparaciones ← comparaciones + c
    intercambios ← intercambios + ic

    regresar (i + 1)

4.1 crearArchivo(resultado).
