2.3 registrarDesdeArchivo:
    selector = new JFileChooser();
    contadorExito = 0;
    contadorFallos = 0;
    Escribir("\nAbriendo selector de archivos...");
    
    si (selector.showOpenDialog(null) == JFileChooser.APPROVE_OPTION) entonces
        archivo = llamar a selector.getSelectedFile();
        intentar
            lector = new FileReader(archivo);
            buffer = new BufferedReader(lector);
            linea = llamar a buffer.readLine();
            
            terminar mientras (linea != null) empezar
                si (!llamar a linea.trim().isEmpty()) entonces
                    nuevaPersona = new Persona();
                    si (llamar a nuevaPersona.setNombre(linea)) entonces
                        llamar a tabla.insertar(nuevaPersona);
                        contadorExito = contadorExito + 1;
                    sino entonces
                       letrero = "Fallo de validacion: "
                        letrero += "Se omitio el registro invalido: ";
                        Escribir(letrero + llamar a linea.trim() + ".");
                        contadorFallos = contadorFallos + 1;
                    terminar si
                terminar si
                linea = llamar a buffer.readLine();
            terminar mientras
            
            llamar a buffer.close();
            Escribir("Proceso finalizado. Se cargaron " + contadorExito 
               + " registros correctamente (" + contadorFallos + " omitidos).");
        capturar excepcion e empezar
            Escribir("Error al leer archivo: " + llamar a e.getMessage());
        terminar intentar
    sino entonces
        Escribir("No se selecciono ningun archivo.");
    terminar si
